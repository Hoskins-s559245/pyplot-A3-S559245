## Question 1 Horizontal Bar Plot

import matplotlib.pyplot as plt

plt.style.use('seaborn-v0_8-darkgrid')
plt.rcParams["figure.figsize"] = [9.50, 7.50]
plt.barh(range(len(letter_freq)), letter_freq.values(), color = 'green', align='center')
plt.yticks(range(len(letter_freq)) , letter_freq.keys())
plt.xlabel('Frequency', fontsize=12)
plt.ylabel('Letters', fontsize=12)


for index, value in enumerate(letter_freq.values()):
    plt.text(value, index,
             str(value))

plt.show()
